# 业务系统传参数据结构

```json
{
  "模型编码": "",       // 模型的唯一编码
  "报价ID": "",        // 业务系统的报价ID
  "报价名称": "",      // 业务系统的报价名称
  "配置信息": {
    "主维度配置": {     // 通常只有主维度的配置
      "维度信息": {    
        "维度1": "",   // 主单据维度1，这些维度是模型已经配好的
        "维度2": "",   // 主单据维度2
        "维度3": "",   // 主单据维度3
        "维度4": "",   // 主单据维度4
        "维度5": ""    // 主单据维度5
        // 可扩展更多维度
      },
      "费用项信息": {
        "费用项编码": "",     // 业务系统的费用项编码
        "费用项序号": "",     // 支持业务系统传多个同名费用项，但是需要通过序号区分开来，如果只有一个，那就是1
        "费用项币种": "",     // 如果是计费规则=实报实销，币种可以传"NONE"，其他的必须传币种
        "是否计费": {         // 决定该费用项是否计费
          "默认计费": "[是/否]",    // 大部分是默认计费的，即匹配到报价就计费；部分费用需要根据单据上的标识判断是否有该费用
          "判断规则": {
            "因子值": "[因子编码]",    // 这个是单据上的某个字段，比如拆板费标识，用来判断单据有没有发生这项费用的；必须要是这个模型关联单据的因子，且必须是引擎因子配置中已经维护的因子。
            "操作符": "[操作符]",      // 必须是引擎中支持的操作符
            "匹配值": "[匹配值]"       // 单据上因子的值跟这个匹配值根据操作符进行判断，条件匹配则计费，否则不计费
          }
        },
        "计费规则": ""        // 计费规则参照计费规则抽象里的计费规则，传参需要按对应计费规则的要求传参
      }
      // 可扩展更多费用项
    },
    "子维度配置": {          // 如异地取货费这种，需要根据单据中的数组字段设置报价的，则按子维度配置传
      "所属主维度": "",       // 关联的主维度，这个主维度就是单据上的数组字段，这个是模型中配置好主维度和子维度关系的
      "维度组合": [          // 多组维度组合
        {
          "维度信息": {          // 这下面是主维度下的子维度信息，这些子维度是模型已经配好的
            "维度1": "",        
            "维度2": ""         
            // 可扩展更多子维度
          },
          "费用项信息": {
            "费用项编码": "",
            "费用项名称": "",
            "费用项序号": "",
            "费用项币种": "",
            "是否计费": {
              "默认计费": "[是/否]",
              "判断规则": {
                "因子值": "[因子编码]",
                "操作符": "[操作符]",
                "匹配值": "[匹配值]"
              }
            },
            "计费规则": ""     
          }
        },
        // 可扩展更多子维度组合
      ]
    }
    // 可扩展更多子维度配置
  }
}
```
